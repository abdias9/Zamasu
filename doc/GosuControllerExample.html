def initialize
    super 800, 600, nil
    @netclient = Zamasu::Client.new 'localhost', 9009
    @netclient.initialize_attribs ({ 'x' => 300, 'y' => 300 })
    @netclient.start_recv_thread

    @sprite = Array.new
    @sprite[DOWN] = Gosu::Image.load_tiles('sprites/down.png', 15, 23)
    @sprite[UP] = Gosu::Image.load_tiles('sprites/up.png', 15, 23)
    @sprite[LEFT] = Gosu::Image.load_tiles('sprites/left.png', 17, 23)
    @sprite[RIGHT] = Gosu::Image.load_tiles('sprites/right.png', 17, 23)

    @pos = DOWN
end

def update
    @moving = true

    if button_down?(Gosu::KbLeft)
        @netclient.increment_attrib('x', -SPEED)
        @pos = LEFT
    elsif button_down?(Gosu::KbRight)
        @netclient.increment_attrib('x', SPEED)
        @pos = RIGHT
    elsif button_down?(Gosu::KbUp)
        @netclient.increment_attrib('y', -SPEED)
        @pos = UP
    elsif button_down?(Gosu::KbDown)
        @netclient.increment_attrib('y', SPEED)
        @pos = DOWN
    else 
        @moving = false
    end
end
