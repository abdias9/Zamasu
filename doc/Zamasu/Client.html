def initialize(host, port)
    @attributes = Hash.new
    @sout = UDPSocket.new
    @sout.connect host, port
    @sin = UDPSocket.new
    @sin.bind nil, 9010
    @sping = UDPSocket.new 
    @sping.bind nil, 9011
end
